function getCookieValue(name) {
    var cookievalue = "";
    var ca = document.cookie.split(";");
    for(var i=0; i<ca.length; i++) {
		var c = ca[i].trim();
		c = c.replace(/\%20/g," ");
	  	if (c.indexOf(name)==0) {
	  	  	cookievalue = c.substring(name.length,c.length);
	 	}
    }
    return cookievalue;
}
